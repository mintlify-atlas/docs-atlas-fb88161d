### Theme Context

The `ThemeProvider` manages theme state with cookie persistence:

```tsx src/context/theme-provider.tsx
import { createContext, useContext, useEffect, useState, useMemo } from 'react'
import { getCookie, setCookie, removeCookie } from '@/lib/cookies'

type Theme = 'dark' | 'light' | 'system'
type ResolvedTheme = Exclude<Theme, 'system'>

const DEFAULT_THEME = 'system'
const THEME_COOKIE_NAME = 'vite-ui-theme'
const THEME_COOKIE_MAX_AGE = 60 * 60 * 24 * 365 // 1 year

export function ThemeProvider({ children }: ThemeProviderProps) {
  const [theme, _setTheme] = useState<Theme>(
    () => (getCookie(THEME_COOKIE_NAME) as Theme) || DEFAULT_THEME
  )

  // Calculate resolved theme (system → dark/light)
  const resolvedTheme = useMemo((): ResolvedTheme => {
    if (theme === 'system') {
      return window.matchMedia('(prefers-color-scheme: dark)').matches
        ? 'dark'
        : 'light'
    }
    return theme as ResolvedTheme
  }, [theme])

  const setTheme = (theme: Theme) => {
    setCookie(THEME_COOKIE_NAME, theme, THEME_COOKIE_MAX_AGE)
    _setTheme(theme)
  }

  // Apply theme to document root
  useEffect(() => {
    const root = window.document.documentElement
    root.classList.remove('light', 'dark')
    root.classList.add(resolvedTheme)
  }, [theme, resolvedTheme])

  return (
    <ThemeContext value={{ theme, resolvedTheme, setTheme }}>
      {children}
    </ThemeContext>
  )
}
```

<Note>
The theme preference is stored in a cookie for persistence across sessions.
</Note>

### Using the Theme Hook

```tsx Using useTheme
import { useTheme } from '@/context/theme-provider'

function ThemeSwitcher() {
  const { theme, setTheme, resolvedTheme } = useTheme()
  
  return (
    <div>
      <p>Current theme: {theme}</p>
      <p>Resolved theme: {resolvedTheme}</p>
      
      <button onClick={() => setTheme('light')}>Light</button>
      <button onClick={() => setTheme('dark')}>Dark</button>
      <button onClick={() => setTheme('system')}>System</button>
    </div>
  )
}
```

## Customizing Colors

To customize your theme colors:

<Steps>
  <Step title="Choose Your Colors">
    Use the [OKLCH Color Picker](https://oklch.com/) to select colors:
    - Adjust Lightness (0-1)
    - Adjust Chroma (saturation, 0-0.4)
    - Adjust Hue (0-360 degrees)
  </Step>
  
  <Step title="Update theme.css">
    Replace the color values in `src/styles/theme.css`:
    ```css
    :root {
      --primary: oklch(0.5 0.2 250);  /* Your custom primary */
    }
    ```
  </Step>
  
  <Step title="Update Dark Mode">
    Adjust dark mode colors for proper contrast:
    ```css
    .dark {
      --primary: oklch(0.7 0.2 250);  /* Lighter for dark mode */
    }
    ```
  </Step>
  
  <Step title="Test Accessibility">
    Ensure sufficient contrast ratios (4.5:1 for text, 3:1 for UI).
  </Step>
</Steps>

<Warning>
Always test color changes in both light and dark modes to ensure readability.
</Warning>

## Typography System

The theme supports custom font configuration:

### Font Provider

```tsx src/context/font-provider.tsx
import { createContext, useContext, useState } from 'react'
import { getCookie, setCookie } from '@/lib/cookies'

type Font = 'inter' | 'manrope'

const DEFAULT_FONT = 'inter'
const FONT_COOKIE_NAME = 'font-family'

export function FontProvider({ children }: { children: React.ReactNode }) {
  const [font, _setFont] = useState<Font>(
    () => (getCookie(FONT_COOKIE_NAME) as Font) || DEFAULT_FONT
  )

  const setFont = (font: Font) => {
    setCookie(FONT_COOKIE_NAME, font, 60 * 60 * 24 * 365)
    _setFont(font)
    document.body.classList.remove('font-inter', 'font-manrope')
    document.body.classList.add(`font-${font}`)
  }

  return (
    <FontContext value={{ font, setFont }}>
      {children}
    </FontContext>
  )
}
```

### Font Configuration

Fonts are defined in `theme.css`:

```css Font Variables
@theme inline {
  --font-inter: 'Inter', 'sans-serif';
  --font-manrope: 'Manrope', 'sans-serif';
}
```

### Switching Fonts

```tsx Font Switcher Example
import { useFont } from '@/context/font-provider'

function FontSwitcher() {
  const { font, setFont } = useFont()
  
  return (
    <select value={font} onChange={(e) => setFont(e.target.value)}>
      <option value="inter">Inter</option>
      <option value="manrope">Manrope</option>
    </select>
  )
}
```

## Border Radius

Global border radius is controlled by the `--radius` variable:

```css src/styles/theme.css
:root {
  --radius: 0.625rem;  /* 10px base radius */
}

@theme inline {
  --radius-sm: calc(var(--radius) - 4px);  /* 6px */
  --radius-md: calc(var(--radius) - 2px);  /* 8px */
  --radius-lg: var(--radius);              /* 10px */
  --radius-xl: calc(var(--radius) + 4px);  /* 14px */
}
```

## Best Practices

<Steps>
  <Step title="Use Semantic Color Names">
    Always use semantic tokens like `bg-primary` instead of specific colors:
    ```tsx
    ✓ className="bg-primary"        // Adapts to theme
    ✗ className="bg-blue-500"      // Fixed color
    ```
  </Step>
  
  <Step title="Test Both Themes">
    Always verify components work in both light and dark modes.
  </Step>
  
  <Step title="Maintain Contrast">
    Ensure text meets WCAG accessibility standards (4.5:1 ratio).
  </Step>
  
  <Step title="Use CSS Variables for Custom Colors">
    When adding new colors, define them as CSS variables first.
  </Step>
</Steps>

<Tip>
Use browser DevTools to inspect computed CSS variables and debug theme issues.
</Tip>

## Example: Custom Theme

Here's how to create a custom purple theme:

```css Custom Purple Theme
:root {
  /* Purple primary color */
  --primary: oklch(0.55 0.25 300);  /* Purple */
  --primary-foreground: oklch(1 0 0);  /* White text */
  
  /* Adjusted accent */
  --accent: oklch(0.95 0.05 300);  /* Light purple */
  --accent-foreground: oklch(0.3 0.15 300);  /* Dark purple */
}

.dark {
  /* Brighter purple for dark mode */
  --primary: oklch(0.7 0.25 300);
  --primary-foreground: oklch(0.15 0.05 300);
  
  --accent: oklch(0.25 0.1 300);
  --accent-foreground: oklch(0.9 0.15 300);
}
```
