The Apps page displays a grid of application integrations with filtering, sorting, and connection status management.

## Overview

The Apps component provides a visual interface for managing third-party app integrations with search, filter, and sort capabilities.

**Location:** `src/features/apps/index.tsx`

## Component Structure

The Apps page uses URL state management for filters and sorting:

```tsx src/features/apps/index.tsx
type AppType = 'all' | 'connected' | 'notConnected'

export function Apps() {
  const {
    filter = '',
    type = 'all',
    sort: initSort = 'asc',
  } = route.useSearch()
  const navigate = route.useNavigate()

  const [sort, setSort] = useState(initSort)
  const [appType, setAppType] = useState(type)
  const [searchTerm, setSearchTerm] = useState(filter)

  const filteredApps = apps
    .sort((a, b) =>
      sort === 'asc'
        ? a.name.localeCompare(b.name)
        : b.name.localeCompare(a.name)
    )
    .filter((app) =>
      appType === 'connected'
        ? app.connected
        : appType === 'notConnected'
          ? !app.connected
          : true
    )
    .filter((app) => app.name.toLowerCase().includes(searchTerm.toLowerCase()))

  return (
    <>
      <Header>
        <Search />
        <div className='ms-auto flex items-center gap-4'>
          <ThemeSwitch />
          <ConfigDrawer />
          <ProfileDropdown />
        </div>
      </Header>

      <Main fixed>
        <div>
          <h1 className='text-2xl font-bold tracking-tight'>
            App Integrations
          </h1>
          <p className='text-muted-foreground'>
            Here&apos;s a list of your apps for the integration!
          </p>
        </div>
        {/* Filters and grid */}
      </Main>
    </>
  )
}
```

## URL State Management

Filters and sorting update the URL for shareability:

```tsx
const handleSearch = (e: ChangeEvent<HTMLInputElement>) => {
  setSearchTerm(e.target.value)
  navigate({
    search: (prev) => ({
      ...prev,
      filter: e.target.value || undefined,
    }),
  })
}

const handleTypeChange = (value: AppType) => {
  setAppType(value)
  navigate({
    search: (prev) => ({
      ...prev,
      type: value === 'all' ? undefined : value,
    }),
  })
}

const handleSortChange = (sort: 'asc' | 'desc') => {
  setSort(sort)
  navigate({ search: (prev) => ({ ...prev, sort }) })
}
```

Example URLs:
- All apps: `/apps`
- Connected only: `/apps?type=connected`
- Search with sort: `/apps?filter=slack&sort=desc`

## Customization

### Adding New Apps

Extend the apps array in `apps/data/apps.tsx`:

```tsx
import { IconCustomApp } from '@/assets/brand-icons'

export const apps = [
  // ... existing apps
  {
    name: 'Custom App',
    logo: <IconCustomApp />,
    connected: false,
    desc: 'Description of your custom app integration.',
  },
]
```

### Custom Connection States

Implement different connection statuses:

```tsx
type ConnectionStatus = 'connected' | 'pending' | 'disconnected' | 'error'

const getStatusBadge = (status: ConnectionStatus) => {
  const badges = {
    connected: 'bg-green-50 text-green-700 border-green-300',
    pending: 'bg-yellow-50 text-yellow-700 border-yellow-300',
    disconnected: 'bg-gray-50 text-gray-700 border-gray-300',
    error: 'bg-red-50 text-red-700 border-red-300',
  }
  return badges[status]
}
```

### Changing Grid Layout

Adjust responsive breakpoints:

```tsx
<ul className='grid gap-4 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5'>
  {/* More columns on larger screens */}
</ul>
```

### Custom Card Styles

Modify the app card appearance:

```tsx
<li className='group rounded-xl border-2 p-6 transition-all hover:border-primary hover:shadow-lg'>
  <div className='mb-6 flex items-center justify-between'>
    <div className='flex size-12 items-center justify-center rounded-xl bg-gradient-to-br from-primary/20 to-primary/10 p-3'>
      {app.logo}
    </div>
    <Button
      variant={app.connected ? 'default' : 'outline'}
      size='sm'
    >
      {app.connected ? 'Disconnect' : 'Connect'}
    </Button>
  </div>
  <div>
    <h2 className='mb-2 text-lg font-semibold'>{app.name}</h2>
    <p className='line-clamp-3 text-sm text-muted-foreground'>{app.desc}</p>
  </div>
</li>
```

### Adding App Categories

Group apps by category:

```tsx
const categories = ['Communication', 'Productivity', 'Business']

{categories.map((category) => (
  <div key={category}>
    <h2 className='mb-4 text-xl font-semibold'>{category}</h2>
    <ul className='grid gap-4 md:grid-cols-2 lg:grid-cols-3'>
      {filteredApps
        .filter((app) => app.category === category)
        .map((app) => (
          <li key={app.name}>{/* App card */}</li>
        ))}
    </ul>
  </div>
))}
```
