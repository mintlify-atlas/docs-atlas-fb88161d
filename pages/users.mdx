The Users page provides a comprehensive user management interface with an advanced data table featuring sticky columns, URL-synced filters, and bulk actions.

## Overview

The Users component demonstrates advanced table features including sticky columns for better horizontal scrolling, role-based filtering, and status management.

**Location:** `src/features/users/index.tsx`

## Table with Sticky Columns

The UsersTable implements sticky columns for better UX on smaller screens:

```tsx src/features/users/components/users-table.tsx
export function UsersTable({ data, search, navigate }: DataTableProps) {
  const [rowSelection, setRowSelection] = useState({})
  const [columnVisibility, setColumnVisibility] = useState<VisibilityState>({})
  const [sorting, setSorting] = useState<SortingState>([])

  // URL state synchronization
  const {
    columnFilters,
    onColumnFiltersChange,
    pagination,
    onPaginationChange,
    ensurePageInRange,
  } = useTableUrlState({
    search,
    navigate,
    pagination: { defaultPage: 1, defaultPageSize: 10 },
    globalFilter: { enabled: false },
    columnFilters: [
      { columnId: 'username', searchKey: 'username', type: 'string' },
      { columnId: 'status', searchKey: 'status', type: 'array' },
      { columnId: 'role', searchKey: 'role', type: 'array' },
    ],
  })

  const table = useReactTable({
    data,
    columns,
    state: { sorting, pagination, rowSelection, columnFilters, columnVisibility },

    getCoreRowModel: getCoreRowModel(),
    getFilteredRowModel: getFilteredRowModel(),
    getPaginationRowModel: getPaginationRowModel(),
    getSortedRowModel: getSortedRowModel(),
    getFacetedRowModel: getFacetedRowModel(),
    getFacetedUniqueValues: getFacetedUniqueValues(),
  })

  return (
    <div className='flex flex-1 flex-col gap-4'>
      <DataTableToolbar
        table={table}
        searchPlaceholder='Filter users...'
        searchKey='username'
        filters={[
          {
            columnId: 'status',
            title: 'Status',
            options: [
              { label: 'Active', value: 'active' },
              { label: 'Inactive', value: 'inactive' },
              { label: 'Invited', value: 'invited' },
              { label: 'Suspended', value: 'suspended' },
            ],
          },
          {
            columnId: 'role',
            title: 'Role',
            options: roles,
          },
        ]}
      />
      {/* Table with sticky columns */}
      <DataTablePagination table={table} />

  )
}
```

export const usersColumns: ColumnDef<User>[] = [

  },
  {
    accessorKey: 'phoneNumber',
    header: ({ column }) => (
      <DataTableColumnHeader column={column} title='Phone Number' />
    ),
    cell: ({ row }) => <div>{row.getValue('phoneNumber')}</div>,
    enableSorting: false,
  },
  {
    accessorKey: 'status',

      return (
        <div className='flex space-x-2'>

      const userType = roles.find(({ value }) => value === role)

          {userType.icon && (
            <userType.icon size={16} className='text-muted-foreground' />

## Sticky Column Implementation

The sticky positioning is achieved through metadata and CSS classes:

<Steps>
  <Step title="Define column metadata">
    Add `className` to the column's `meta` property with sticky positioning classes:
    ```tsx
    meta: {
      className: cn('max-md:sticky start-0 z-10'),
    }
    ```
  </Step>
  
  <Step title="Apply classes to table cells">
    The table renderer applies these classes to both header and body cells:
    ```tsx
    <TableHead
      className={cn(
        'bg-background group-hover/row:bg-muted',
        header.column.columnDef.meta?.className,
        header.column.columnDef.meta?.thClassName
      )}
    >
    ```
  </Step>
  
  <Step title="Add background colors">
    Ensure sticky cells have background colors that update on hover:
    ```tsx
    className={cn(
      'bg-background group-hover/row:bg-muted group-data-[state=selected]/row:bg-muted',
      cell.column.columnDef.meta?.className
    )}
    ```
  </Step>
</Steps>

## Filtering Options

The users table supports three types of filters:

### Username Filter (String)

Text-based search for usernames:

```tsx
{ columnId: 'username', searchKey: 'username', type: 'string' }
```

URL: `?username=john`

### Status Filter (Array)

Multi-select filter for user status:

```tsx
{
  columnId: 'status',
  title: 'Status',
  options: [
    { label: 'Active', value: 'active' },
    { label: 'Inactive', value: 'inactive' },
    { label: 'Invited', value: 'invited' },
    { label: 'Suspended', value: 'suspended' },
  ],
}
```

URL: `?status=active&status=invited`

### Role Filter (Array)

Multi-select filter for user roles:

```tsx
{
  columnId: 'role',
  title: 'Role',
  options: roles.map((role) => ({ ...role })),
}
```

URL: `?role=admin&role=user`

## Customization

### Adding New Columns

Extend the `usersColumns` array:

```tsx
{
  accessorKey: 'department',
  header: ({ column }) => (
    <DataTableColumnHeader column={column} title='Department' />
  ),
  cell: ({ row }) => <div>{row.getValue('department')}</div>,
}
```

### Customizing Sticky Behavior

Modify sticky column classes:

```tsx
meta: {
  // Always sticky (not just on mobile)
  className: cn('sticky start-0 z-10'),
}
```

### Adding Custom Status Colors

Define custom badge colors in `data/data.ts`:

```tsx
export const callTypes = new Map<string, string>([
  ['active', 'border-green-600 text-green-600'],
  ['inactive', 'border-gray-400 text-gray-400'],
  ['invited', 'border-blue-600 text-blue-600'],
  ['suspended', 'border-red-600 text-red-600'],
])
```
